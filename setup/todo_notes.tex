\newcommand{\namedtodo}[5]
{
  \stepcounter{todofull}
  \ifthenelse{\equal{#1}{}}
  {
    \todo[backgroundcolor=#4,linecolor=#4,caption=
    {\textbf{\arabic{todofull} #3: } #2}]
    {\color{#5}\textbf{\arabic{todofull} #3: }#2}
  }
  {
    \todo[backgroundcolor=#4,linecolor=#4,caption=
    {\textbf{\arabic{todofull} #3: } #1}]
    {\color{#5}\textbf{\arabic{todofull} #3: }#2}
  }
}

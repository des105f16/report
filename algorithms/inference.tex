\begin{algorithm}[h]
\SetAlgoNoEnd
CHECK\_LABELS(p)\\
\ForEach {$c \in Q$}
{
  \If {left($c$) is variable}
  {
    upper($c$) = $\top$
  }
  \Else
  {
    upper($c$) = left($c$)
  }
}
\While {$\exists c \in Q | upper(c) \sqsubseteq right(c)$}
{
  pick $c \in Q | upper(c) \sqsubseteq right(c)$\\
  \If {left($c$) is not variable}
  {
    $ERROR$
  }
  let upper($c$) = upper($c$) $\sqcup$ right($c$)
}
\end{algorithm}
